local function AutoChat(msg)
    task.spawn(function()
        local TextChatService = game:GetService("TextChatService")
        if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
            local channel = TextChatService.TextChannels.RBXGeneral
            if channel then
                channel:SendAsync(msg)
            end
        end
        
        local chatEvents = game:GetService("ReplicatedStorage"):FindFirstChild("DefaultChatSystemChatEvents")
        if chatEvents and chatEvents:FindFirstChild("SayMessageRequest") then
            chatEvents.SayMessageRequest:FireServer(msg, "All")
        end
    end)
end

local function flingPlayer(targetPlayer, statusDot)
    local char = player.Character
    local lrp = char and char:FindFirstChild("HumanoidRootPart")
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    local targetChar = targetPlayer.Character
    local trp = targetChar and targetChar:FindFirstChild("HumanoidRootPart")
    local targetHum = targetChar and targetChar:FindFirstChildOfClass("Humanoid")
    
    if lrp and trp and hum and targetHum then
        IsFlinging = true
        statusDot.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        local oldPos = lrp.CFrame
        camera.CameraSubject = targetHum
        
        local bav = Instance.new("BodyAngularVelocity", lrp)
        bav.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
        bav.AngularVelocity = Vector3.new(1000000, 1000000, 1000000) 
        local bv = Instance.new("BodyVelocity", lrp)
        bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
        bv.Velocity = Vector3.new(1000000, 1000000, 1000000)

        local connection
        local timer = 0
        connection = RunService.Heartbeat:Connect(function()
            if targetChar and trp and targetHum.Health > 0 then
                timer = timer + 0.6
                local ySweep = math.sin(timer * 2.5) * 4 
                local xSweep = math.cos(timer * 2) * 3  
                local zSweep = math.sin(timer * 1.5) * 5 
                lrp.CFrame = trp.CFrame * CFrame.new(xSweep, ySweep, zSweep) * CFrame.Angles(math.rad(90), math.random(-20, 20), math.random(-20, 20))
                lrp.AssemblyLinearVelocity = Vector3.new(1000000, 1000000, 1000000)
                for _, part in pairs(char:GetDescendants()) do
                    if part:IsA("BasePart") then part.CanCollide = false end
                end
            else
                connection:Disconnect()
            end
        end)
        
        task.wait(1.2)
        connection:Disconnect()
        bav:Destroy()
        bv:Destroy()
        lrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        lrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
        for _, part in pairs(char:GetDescendants()) do if part:IsA("BasePart") then part.CanCollide = true end end
        lrp.CFrame = oldPos
        camera.CameraSubject = hum
        hum:ChangeState(Enum.HumanoidStateType.GettingUp)
        task.wait(0.1)
        IsFlinging = false
        statusDot.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    end
end

local function updateList()
    for _, child in pairs(ScrollingFrame:GetChildren()) do if child:IsA("TextButton") then child:Destroy() end end
    for _, p in pairs(game.Players:GetPlayers()) do
        if p ~= player then
            local btn = Instance.new("TextButton", ScrollingFrame)
            btn.Size = UDim2.new(1, -5, 0, 35)
            btn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
            btn.Text = "   " .. (p.DisplayName or p.Name)
            btn.TextColor3 = Color3.fromRGB(255, 255, 255)
            btn.Font = Enum.Font.GothamMedium
            btn.TextSize = 11
            btn.TextXAlignment = Enum.TextXAlignment.Left
            btn.ZIndex = 4
            Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)
            
            local StatusDot = Instance.new("Frame", btn)
            StatusDot.Size = UDim2.new(0, 8, 0, 8)
            StatusDot.Position = UDim2.new(1, -20, 0.5, -4)
            StatusDot.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
            StatusDot.ZIndex = 5
            Instance.new("UICorner", StatusDot).CornerRadius = UDim.new(1, 0)
            
            btn.MouseButton1Click:Connect(function() if not IsFlinging then flingPlayer(p, StatusDot) end end)
        end
    end
end
