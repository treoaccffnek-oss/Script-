print("--- START HYDRADAT GG2 RAW ---")

-- Tạo một dòng chữ thô sơ ngay giữa màn hình để kiểm tra xem script có lên hay không
local hint = Instance.new("Hint")
hint.Parent = game:GetService("CoreGui")
hint.Text = "TIKTOK: HYDRADAT - SCRIPT DANG CHAY CHONG LOI"

local Workspace = game:GetService("Workspace")
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local player = Players.LocalPlayer

local lastAction = tick()

-- Hàm kích hoạt nút bấm siêu tốc không dùng vòng lặp quét sâu
function RunFarm()
    local char = player.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    -- Kiểm tra ngày đêm dựa trên độ sáng của game
    local isNight = Lighting.OutdoorAmbient.R < 55

    -- 1. QUÈT NHANH TOÀN MAP ĐỂ HÁI VÀ TRỒNG SPEED CÓ SẴN TRONG KHO
    local descendants = Workspace:GetDescendants()
    for i = 1, #descendants do
        local v = descendants[i]
        if v:IsA("ProximityPrompt") then
            local txt = string.lower(v.ActionText .. v.ObjectText .. v.Name)
            
            -- Tự động hái quả liên tục
            if string.find(txt, "harv") or string.find(txt, "hái") then
                fireproximityprompt(v)
            end
            
            -- Tự động trồng hạt thường và hạt Speed khi có ô đất trống
            if string.find(txt, "plant") or string.find(txt, "trồng") or string.find(txt, "dirt") then
                fireproximityprompt(v)
            end
        end
    end

    -- 2. CHU KỲ BAN NGÀY: CỨ 10 GIÂY TELE ĐI BÁN VÀ MUA
    if not isNight then
        if tick() - lastAction >= 10 then
            local savedPos = hrp.CFrame

            -- Biến đến bục Bán (Sell)
            for i = 1, #descendants do
                local v = descendants[i]
                if v:IsA("ProximityPrompt") and string.find(string.lower(v.ActionText .. v.Name), "sell") then
                    if v.Parent:IsA("BasePart") then
                        hrp.CFrame = CFrame.new(v.Parent.Position + Vector3.new(0, 2, 0))
                        task.wait(0.4)
                        fireproximityprompt(v)
                        task.wait(0.4)
                        break
                    end
                end
            end

            -- Biến đến bục Mua (Speed/Shop)
            for i = 1, #descendants do
                local v = descendants[i]
                if v:IsA("ProximityPrompt") and (string.find(string.lower(v.ActionText .. v.Name), "speed") or string.find(string.lower(v.ActionText .. v.Name), "buy")) then
                    if v.Parent:IsA("BasePart") then
                        hrp.CFrame = CFrame.new(v.Parent.Position + Vector3.new(0, 2, 0))
                        task.wait(0.4)
                        fireproximityprompt(v)
                        task.wait(0.4)
                        break
                    end
                end
            end

            -- Quay trở lại vườn cũ
            hrp.CFrame = savedPos
            lastAction = tick()
        end
    else
        -- BAN ĐÊM: Khóa vị trí nhân vật tại vườn để thủ quả, chống tele đi bán lung tung
        for i = 1, #descendants do
            local v = descendants[i]
            if v:IsA("BasePart") and (string.find(string.lower(v.Name), "plot") or string.find(string.lower(v.Name), "dirt")) then
                hrp.CFrame = CFrame.new(v.Position + Vector3.new(0, 4, 0))
                break
            end
        end
    end
end

-- Vòng lặp thô sơ ép Executor phải chạy liên tục mỗi 0.5 giây
while true do
    pcall(RunFarm)
    task.wait(0.5)
end
