-- CHỨC NĂNG: ĐỨNG IM MỘT CHỖ TỰ HÚT - TỰ BÁN - TỰ MUA - TỰ TRỒNG TOÀN MAP
-- CAM KẾT: KHÔNG TELEPORT CHỐNG BAN ACC 100% - KHÔNG CÓ CHỮ CHE MÀN HÌNH
-- =========================================================================

-- XÓA SẠCH TẤT CẢ CÁC LOẠI CHỮ NỀN TRẮNG/CẦU VỒNG CŨ
for _, v in pairs(game:GetService("CoreGui"):GetChildren()) do
    if v:IsA("Hint") or v.Name == "HydraDat_Watermark" then v:Destroy() end
end

local Workspace = game:GetService("Workspace")
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local player = Players.LocalPlayer

task.spawn(function()
    while true do
        task.wait(0.3) -- Tốc độ quét tối ưu để hút đồ liên tục không ngừng nghỉ
        
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        
        if hrp then
            pcall(function()
                -- Nhận diện trời tối hoặc cầu vồng để tăng tốc nhặt
                local isNightOrRainbow = (Lighting.OutdoorAmbient.R < 60) or (Lighting.ClockTime >= 18 or Lighting.ClockTime <= 6)
                local allParts = Workspace:GetDescendants()

                -- 1. HACK BÁN KÍNH: ÉP TẤT CẢ NÚT BẤM TOÀN MAP BAY VỀ TAY BẠN (XÓA CHỜ 0S)
                for i = 1, #allParts do
                    local v = allParts[i]
                    if v:IsA("ProximityPrompt") then
                        v.HoldDuration = 0 -- Bỏ qua thời gian ấn giữ nút
                        v.MaxActivationDistance = 999999 -- Khoảng cách vô hạn, đứng góc map vẫn bấm được nút ngoài chợ
                        v.RequiresLineOfSight = false -- Bỏ qua vật cản, xuyên tường luôn
                    end
                end

                -- 2. TỰ ĐỘNG HÁI QUẢ VÀ NHẶT HẠT GIỐNG RỚT DƯỚI ĐẤT TOÀN SERVER
                for i = 1, #allParts do
                    local v = allParts[i]
                    if v:IsA("ProximityPrompt") then
                        local txt = string.lower(v.ActionText .. v.ObjectText .. v.Name)
                        
                        -- Hút quả chín toàn map
                        if string.find(txt, "harv") or string.find(txt, "hái") then
                            fireproximityprompt(v)
                        end
                        
                        -- Hút hạt giống rơi vãi khắp bản đồ
                        if string.find(txt, "collect") or string.find(txt, "take") or string.find(txt, "seed") or string.find(txt, "nhặt") then
                            fireproximityprompt(v)
                        end
                    end
                end

                -- 3. CHU KỲ BÁN / MUA / TRỒNG (KHÔNG DI CHUYỂN)
                if not isNightOrRainbow then
                    -- BAN NGÀY: Đứng im tại chỗ tự động kích hoạt nút Bán ngoài chợ để xả kho đồ đầy
                    for i = 1, #allParts do
                        local v = allParts[i]
                        if v:IsA("ProximityPrompt") and (string.find(string.lower(v.ActionText), "sell") or string.find(string.lower(v.Name), "sell")) then
                            fireproximityprompt(v)
                        end
                    end

                    -- Đứng im tại chỗ tự động kích hoạt nút Mua hạt Speed ngoài Shop
                    for i = 1, #allParts do
                        local v = allParts[i]
                        if v:IsA("ProximityPrompt") and (string.find(string.lower(v.ActionText), "speed") or string.find(string.lower(v.Name), "shop") or string.find(string.lower(v.ActionText), "buy")) then
                            fireproximityprompt(v)
                        end
                    end
                end

                -- LUÔN LUÔN TỰ ĐỘNG TRỒNG (Dù ngày hay đêm, cứ đất trống là gieo hạt giống/Speed liền)
                for i = 1, #allParts do
                    local v = allParts[i]
                    if v:IsA("ProximityPrompt") then
                        local txt = string.lower(v.ActionText .. v.ObjectText)
                        if string.find(txt, "plant") or string.find(txt, "trồng") or string.find(txt, "dirt") then
                            fireproximityprompt(v)
                        end
                    end
                end

            end)
        end
    end
end)
