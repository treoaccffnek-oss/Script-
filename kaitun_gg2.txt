-- =========================================================================
-- SCRIPT: HYDRADAT ANTI-BAN STEALTH RUN - GROW A GARDEN 2
-- CHỨC NĂNG: BỎ HOÀN TOÀN TELEPORT CHỐNG BAN - CHỈ CHẠY BỘ - HÚT ĐỒ TỪ XA 0S
-- =========================================================================

for _, v in pairs(game:GetService("CoreGui"):GetChildren()) do
    if v:IsA("Hint") or v.Name == "HydraDat_Watermark" then v:Destroy() end
end

local Workspace = game:GetService("Workspace")
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local player = Players.LocalPlayer

local testHint = Instance.new("Hint", game:GetService("CoreGui"))
testHint.Text = "=== HYDRADAT GG2: CHEDO AN TOAN (ANTI-BAN) DANG CHAY ==="

task.spawn(function()
    while true do
        task.wait(0.3) -- Giữ nhịp quét mượt mà, không làm nghẽn game
        
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if hrp and hum then
            pcall(function()
                -- Nhận diện trời tối hoặc cầu vồng
                local isNightOrRainbow = (Lighting.OutdoorAmbient.R < 60) or (Lighting.ClockTime >= 18 or Lighting.ClockTime <= 6)
                local allParts = Workspace:GetDescendants()

                -- 1. BẺ KHÓA NÚT BẤM: GIỮ = 0 GIÂY VÀ TĂNG PHẠM VI TƯƠNG TÁC TỪ XA
                for i = 1, #allParts do
                    local v = allParts[i]
                    if v:IsA("ProximityPrompt") then
                        v.HoldDuration = 0 -- Bấm phát ăn ngay không cần chờ vòng tròn xoay
                        v.MaxActivationDistance = 40 -- Đứng từ xa (40m) vẫn bấm và nhặt được đồ
                    end
                end

                -- 2. TỰ ĐỘNG ĐIỀU CHỈNH TỐC ĐỘ CHẠY BỘ (AN TOÀN)
                if isNightOrRainbow then
                    hum.WalkSpeed = 30 -- Tốc độ chạy nhanh vừa phải lúc đêm/cầu vồng để quét hạt
                else
                    hum.WalkSpeed = 16 -- Ban ngày chạy bình thường
                end

                -- 3. VÒNG LẶP HÚT ĐỒ TỰ ĐỘNG (KHÔNG DÙNG TELEPORT CFRAME)
                for i = 1, #allParts do
                    local v = allParts[i]
                    if v:IsA("ProximityPrompt") then
                        local txt = string.lower(v.ActionText .. v.ObjectText .. v.Name)
                        
                        -- Hái quả chín từ xa
                        if string.find(txt, "harv") or string.find(txt, "hái") then
                            fireproximityprompt(v)
                        end
                        
                        -- Hút sạch hạt giống rớt dưới đất từ xa (Bán kính 40m nhờ MaxActivationDistance)
                        if string.find(txt, "collect") or string.find(txt, "take") or string.find(txt, "seed") or string.find(txt, "nhặt") then
                            fireproximityprompt(v)
                        end
                        
                        -- Tự động trồng hạt thường/Speed trong kho xuống đất trống
                        if string.find(txt, "plant") or string.find(txt, "trồng") or string.find(txt, "dirt") then
                            fireproximityprompt(v)
                        end
                    end
                end

                -- 4. BAN NGÀY TỰ CHẠY BỘ RA KHU SHOP (KHÔNG TELE ĐỂ CHỐNG KẸT/CHỐNG BAN)
                if not isNightOrRainbow then
                    for i = 1, #allParts do
                        local v = allParts[i]
                        if v:IsA("ProximityPrompt") then
                            local txt = string.lower(v.ActionText .. v.Name)
                            
                            -- Nếu đi ngang qua bục Bán đồ, tự động xả kho
                            if string.find(txt, "sell") or string.find(txt, "bán") then
                                fireproximityprompt(v)
                            end
                            
                            -- Nếu đi ngang qua bục Mua đồ, tự động gom thêm hạt Speed
                            if string.find(txt, "speed") or string.find(txt, "shop") or string.find(txt, "buy") or string.find(txt, "mua") then
                                fireproximityprompt(v)
                            end
                        end
                    end
                end
                
            end)
        end
    end
end)
